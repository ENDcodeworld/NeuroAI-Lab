### 3.1 EEG 信号处理模块

```python
# 伪代码示例
class EEGProcessor:
    def __init__(self, sample_rate: int = 256):
        self.sample_rate = sample_rate
        self.filters = self._init_filters()
    
    def preprocess(self, raw_data: np.ndarray) -> np.ndarray:
        """预处理：滤波、去噪、伪迹去除"""
        # 1. 带通滤波 (0.5-45Hz)
        filtered = self._bandpass_filter(raw_data, 0.5, 45)
        
        # 2. 工频滤波 (50Hz)
        filtered = self._notch_filter(filtered, 50)
        
        # 3. 伪迹去除 (眼动、肌电)
        cleaned = self._remove_artifacts(filtered)
        
        return cleaned
    
    def extract_features(self, eeg_data: np.ndarray) -> dict:
        """特征提取"""
        features = {
            'power_spectrum': self._calc_power_spectrum(eeg_data),
            'band_power': self._calc_band_power(eeg_data),  # δθ α β γ
            'coherence': self._calc_coherence(eeg_data),
            'microstates': self._calc_microstates(eeg_data),
        }
        return features
    
    def _calc_band_power(self, eeg_data: np.ndarray) -> dict:
        """计算各频段功率"""
        bands = {
            'delta': (0.5, 4),
            'theta': (4, 8),
            'alpha': (8, 13),
            'beta': (13, 30),
            'gamma': (30, 45),
        }
        return {
            band: self._power_in_range(eeg_data, low, high)
            for band, (low, high) in bands.items()
        }
```

### 3.2 AI 分析模型

#### 3.2.1 专注度评估模型

```python
class AttentionModel(nn.Module):
    """基于 EEG 的专注度评估"""
    
    def __init__(self, num_channels: int = 14):
        super().__init__()
        self.conv1d = nn.Conv1d(num_channels, 64, kernel_size=3)
        self.lstm = nn.LSTM(64, 128, batch_first=True)
        self.attention = nn.MultiheadAttention(128, 8)
        self.fc = nn.Sequential(
            nn.Linear(128, 64),
            nn.ReLU(),
            nn.Dropout(0.3),
            nn.Linear(64, 1),  # 专注度分数 0-100
            nn.Sigmoid()
        )
    
    def forward(self, x: torch.Tensor) -> torch.Tensor:
        # x: (batch, time, channels)
        x = x.permute(0, 2, 1)  # (batch, channels, time)
        x = self.conv1d(x).permute(0, 2, 1)  # (batch, time, 64)
        x, _ = self.lstm(x)  # (batch, time, 128)
        x, _ = self.attention(x, x, x)
        x = x.mean(dim=1)  # (batch, 128)
        return self.fc(x)
```

```python
class EmotionModel(nn.Module):
    """情绪识别：valence + arousal"""
    
    def __init__(self, num_channels: int = 14):
        super().__init__()
        self.feature_extractor = EEGFeatureExtractor(num_channels)
        self.valence_head = nn.Sequential(
            nn.Linear(256, 128),
            nn.ReLU(),
            nn.Linear(128, 1),
            nn.Sigmoid()
        )
        self.arousal_head = nn.Sequential(
            nn.Linear(256, 128),
            nn.ReLU(),
            nn.Linear(128, 1),
            nn.Sigmoid()
        )
    
    def forward(self, x: torch.Tensor) -> dict:
        features = self.feature_extractor(x)
        return {
            'valence': self.valence_head(features),  # 愉悦度
            'arousal': self.arousal_head(features),  # 唤醒度
        }
```

### 3.3 神经反馈训练引擎

```python
class NeurofeedbackEngine:
    """实时神经反馈训练"""
    
    def __init__(self, target_band: str = 'alpha', threshold: float = 0.7):
        self.target_band = target_band
        self.threshold = threshold
        self.processor = EEGProcessor()
        self.feedback_callback = None
    
    def start_session(self, device: BLEDevice):
        """开始训练会话"""
        device.start_streaming(self._on_eeg_data)
    
    def _on_eeg_data(self, eeg_data: np.ndarray):
        """实时 EEG 数据回调"""
        # 1. 预处理
        cleaned = self.processor.preprocess(eeg_data)
        
        # 2. 提取目标频段功率
        features = self.processor.extract_features(cleaned)
        band_power = features['band_power'][self.target_band]
        
        # 3. 判断是否达标
        success = band_power > self.threshold
        
        # 4. 触发反馈
        if self.feedback_callback:
            self.feedback_callback(success, band_power)
    
    def set_feedback_callback(self, callback: callable):
        """设置反馈回调（用于游戏/视觉反馈）"""
        self.feedback_callback = callback
```

```sql
-- 用户表
CREATE TABLE users (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    email VARCHAR(255) UNIQUE NOT NULL,
    password_hash VARCHAR(255) NOT NULL,
    nickname VARCHAR(100),
    avatar_url TEXT,
    created_at TIMESTAMP DEFAULT NOW(),
    updated_at TIMESTAMP DEFAULT NOW()
);

-- 订阅表
CREATE TABLE subscriptions (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    user_id UUID REFERENCES users(id),
    plan_id VARCHAR(50) NOT NULL,
    status VARCHAR(50) NOT NULL,
    current_period_start TIMESTAMP,
    current_period_end TIMESTAMP,
    created_at TIMESTAMP DEFAULT NOW()
);

-- EEG 会话表
CREATE TABLE eeg_sessions (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    user_id UUID REFERENCES users(id),
    device_type VARCHAR(50),
    duration_seconds INTEGER,
    raw_data_path TEXT,  -- S3 路径
    created_at TIMESTAMP DEFAULT NOW()
);

-- 分析结果表
CREATE TABLE analysis_results (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    session_id UUID REFERENCES eeg_sessions(id),
    metrics JSONB NOT NULL,  -- { alpha, beta, theta, delta, gamma }
    attention_score FLOAT,
    emotion_valence FLOAT,
    emotion_arousal FLOAT,
    created_at TIMESTAMP DEFAULT NOW()
);

-- 训练记录表
CREATE TABLE training_records (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    user_id UUID REFERENCES users(id),
    program_id VARCHAR(50) NOT NULL,
    session_date DATE NOT NULL,
    duration_seconds INTEGER,
    performance_score FLOAT,
    created_at TIMESTAMP DEFAULT NOW()
);
```

## 6. 安全设计

### 6.1 认证与授权

- JWT Token 认证（access_token + refresh_token）
- Token 有效期：access 15 分钟，refresh 7 天
- RBAC 权限控制（user, admin, super_admin）
- API 限流：每用户 100 请求/分钟

### 6.2 数据加密

- 传输加密：HTTPS (TLS 1.3)
- 存储加密：AES-256（敏感字段）
- 密码哈希：bcrypt (cost=12)

### 6.3 隐私保护

- EEG 数据匿名化处理
- 用户数据导出/删除支持（GDPR 合规）
- 数据最小化原则
